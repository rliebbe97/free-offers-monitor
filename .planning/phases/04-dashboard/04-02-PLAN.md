     ```typescript
     const db = createClient();
     let query = db
       .from('offers')
       .select('id, title, status, destination_url, extraction_confidence, created_at', { count: 'exact' });
     
     if (status !== 'all') {
       query = query.eq('status', status);
     }
     
     // Sort mapping
     if (sort === 'oldest') {
       query = query.order('created_at', { ascending: true });
     } else if (sort === 'confidence') {
       query = query.order('extraction_confidence', { ascending: false, nullsFirst: false });
     } else {
       query = query.order('created_at', { ascending: false }); // default: newest
     }
     
     query = query.range(from, to);
     ```

   - Compute `totalPages = Math.ceil((count ?? 0) / PAGE_SIZE)`
